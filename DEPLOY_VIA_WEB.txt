    <script>
        function switchTab(tab) {
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(tc => tc.classList.remove('active'));
            
            event.target.classList.add('active');
            document.getElementById(tab + '-tab').classList.add('active');
        }

        function selectVPN(server) {
            document.querySelectorAll('.server-card').forEach(card => card.classList.remove('active'));
            event.target.closest('.server-card').classList.add('active');
            
            showMessage('wifi', 'VPN CONFIG FOR ' + server.toUpperCase() + ' - USE ADVANCED SETTINGS TO CONFIGURE', 'info');
        }

        function saveWifiConfig() {
            const ssid = document.getElementById('wifi-ssid').value;
            const password = document.getElementById('wifi-password').value;
            const channel = document.getElementById('wifi-channel').value;

            if (!ssid || !password) {
                showMessage('wifi', 'SSID AND PASSWORD REQUIRED', 'error');
                return;
            }

            showMessage('wifi', 'UPDATING WIFI CONFIGURATION...', 'info');

            fetch('/cgi-bin/luci/admin/network/wireless', {
                method: 'POST',
                headers: {'Content-Type': 'application/x-www-form-urlencoded'},
                body: `ssid=${encodeURIComponent(ssid)}&key=${encodeURIComponent(password)}&channel=${channel}`
            })
            .then(() => {
                showMessage('wifi', 'WIFI UPDATED - RECONNECT TO NEW NETWORK', 'success');
            })
            .catch(() => {
                showMessage('wifi', 'USE ADVANCED SETTINGS FOR WIFI CONFIG', 'info');
            });
        }

        function scanNetworks() {
            showMessage('portal', 'SCANNING NETWORKS...', 'info');
            
            fetch('/cgi-bin/luci/admin/network/wireless_scan_trigger')
                .then(r => r.json())
                .then(data => {
                    const select = document.getElementById('public-networks');
                    select.innerHTML = '<option value="">SELECT NETWORK</option>';
                    
                    if (data && data.results) {
                        data.results.forEach(net => {
                            select.innerHTML += `<option value="${net.ssid}">${net.ssid} (${net.signal}%)</option>`;
                        });
                    }
                    
                    showMessage('portal', 'SCAN COMPLETE', 'success');
                })
                .catch(() => {
                    showMessage('portal', 'USE ADVANCED SETTINGS TO CONNECT TO PUBLIC WIFI', 'info');
                });
        }

        function openCaptivePortal() {
            const network = document.getElementById('public-networks').value;
            
            if (!network) {
                showMessage('portal', 'SELECT A NETWORK FIRST', 'error');
                return;
            }

            showMessage('portal', 'OPENING CAPTIVE PORTAL...', 'info');
            window.open('http://detectportal.firefox.com/', '_blank');
        }

        function showMessage(section, text, type) {
            const msg = document.getElementById(section + '-message');
            msg.textContent = text;
            msg.className = 'message show';
            setTimeout(() => msg.classList.remove('show'), 5000);
        }
